        <Link
          href={page.route}
          style={{ color: 'inherit', textDecoration: 'none' }}
          className="block hover:border-primary border border-transparent rounded-tl-lg rounded-br-lg mx-8 lg:mx-16 p-2"
          tabIndex="0"
        >
        <div className="w-full h-[30vh] p-8 relative rounded">
            <Image
              src={page.frontMatter.image}
              fill
              className="p-2 rounded-[12px]"
            />
          </div>
        <div className="flex">
          <div>
            <span className="text-base md:text-xl lg:text-2xl font-bold">{page.frontMatter.title}</span>    
            <p className="opacity-80 line-clamp-3 m-1 text-sm md:text-base ">
              {page.frontMatter?.description}{' '}
            </p>
            {page.frontMatter?.date ? (
              <p className=" text-xs md:text-sm my-2 text-primary font-medium">{page.frontMatter.tags}</p>
            ) : null} 
            {page.frontMatter?.date ? (
              <p className="opacity-40 text-xs md:text-sm my-2">{page.frontMatter.date}</p>
            ) : null} 
          </div>
        </div>
        </Link>
      </div>
    );
  });
}
